class GreatestCommonDivisor {
	int a = 10;
	int b = 20;
	bool x[10];
	void main() {
		int w, y, z;
		read w;
		read y;
		z = gcd(w, y);

		print "gcd (", w, ",", y, ")=", z, "\n";
		w = 4;
		for(w=1; w<10; w=w+1) {
			read x[w];
		}
		w = 4;
		for(w = 1; w<10; w=w+1) {
			print x[w] || true, "\n";
		}
	}

	// Función que calcula el máximo común divisor
	int gcd(int a, int b) {
		if (b == 0) { 
			return(a); 
		}
		else {
			return gcd(b, a % b);
		}
	}
}
